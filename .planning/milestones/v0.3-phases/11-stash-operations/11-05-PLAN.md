6. **Add makeStashItem function** (after `makeWipItem`, around line 120):
```typescript
function makeStashItem(stash: StashEntry, stashCol: number, colorIdx: number): GraphCommit {
  // Find the parent commit's column so the fork edge connects to the right place
  const parentCommit = commits.find(c => c.oid === stash.parent_oid);
  const parentCol = parentCommit ? parentCommit.column : stashCol;

  // Determine edge type based on relative column positions
  let edgeType: EdgeType;
  if (stashCol === parentCol) {
    edgeType = 'Straight';
  } else if (stashCol > parentCol) {
    edgeType = 'ForkLeft';   // stash is to the right, fork goes left-down to parent
  } else {
    edgeType = 'ForkRight';  // stash is to the left, fork goes right-down to parent
  }

  return {
    oid: `__stash_${stash.index}__`,
    short_oid: '',
    summary: stash.name,
    body: null,
    author_name: '',
    author_email: '',
    author_timestamp: 0,
    parent_oids: stash.parent_oid ? [stash.parent_oid] : [],
    column: stashCol,
    color_index: colorIdx,
    edges: [{
      from_column: stashCol,
      to_column: parentCol,
      edge_type: edgeType,
      color_index: colorIdx,
    }],
    refs: [],
    is_head: false,
    is_merge: false,
    is_branch_tip: true,  // no incoming rail from above
  };
}
```

8. **Replace displayItems** (line 122-126) with a version that injects stash rows:
```typescript
const displayItems = $derived.by(() => {
  // Start with base items (WIP row + commits)
  const base: GraphCommit[] = wipCount > 0
    ? [makeWipItem(wipMessage), ...commits]
    : [...commits];

  if (stashes.length === 0) return base;

  // Insert each stash row immediately ABOVE its parent commit
  // Process in reverse order so splice indices remain stable
  const result = [...base];
  for (let i = stashes.length - 1; i >= 0; i--) {
    const stash = stashes[i];
    if (!stash.parent_oid) continue;

    const parentIdx = result.findIndex(c => c.oid === stash.parent_oid);
    if (parentIdx === -1) continue; // parent not in current page

    const colorIdx = (stashColumn + i) % 8;
    const stashItem = makeStashItem(stash, stashColumn, colorIdx);
    // Insert BEFORE the parent commit (stash row appears above its parent)
    result.splice(parentIdx, 0, stashItem);
  }

  return result;
});
```

Note: Use `$derived.by(() => { ... })` which is the Svelte 5 syntax for derived values with function bodies. Do NOT use `$derived(() => { ... })()` IIFE pattern (that was used in 11-02 and may have contributed to issues).

```typescript
async function showStashContextMenu(e: MouseEvent, stashIndex: number) {
  e.preventDefault();
  e.stopPropagation(); // prevent CommitRow onclick from firing
  const menu = await Menu.new({
    items: [
      await MenuItem.new({
        text: 'Pop',
        action: async () => {
          try {
            await safeInvoke('stash_pop', { path: repoPath, index: stashIndex });
            stashError = null;
          } catch (err) {
            stashError = (err as TrunkError).message ?? 'Failed to pop stash';
          }
        },
      }),
      await MenuItem.new({
        text: 'Apply',
        action: async () => {
          try {
            await safeInvoke('stash_apply', { path: repoPath, index: stashIndex });
            stashError = null;
          } catch (err) {
            stashError = (err as TrunkError).message ?? 'Failed to apply stash';
          }
        },
      }),
      await MenuItem.new({
        text: 'Drop',
        action: async () => {
          try {
            const { ask } = await import('@tauri-apps/plugin-dialog');
            const confirmed = await ask(
              `Drop stash@{${stashIndex}}? This cannot be undone.`,
              { title: 'Confirm Drop', kind: 'warning' }
            );
            if (!confirmed) return;
            await safeInvoke('stash_drop', { path: repoPath, index: stashIndex });
            stashError = null;
          } catch (err) {
            stashError = (err as TrunkError).message ?? 'Failed to drop stash';
          }
        },
      }),
    ],
  });
  await menu.popup();
}
```

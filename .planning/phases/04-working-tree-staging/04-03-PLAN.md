```typescript
$effect(() => {
  let unlisten: (() => void) | undefined;
  listen<string>('repo-changed', (event) => {
    if (event.payload === repoPath) loadStatus();
  }).then((fn) => { unlisten = fn; });
  return () => { unlisten?.(); };
});
```

   ```typescript
   function startColumnResize(column: keyof ColumnWidths, e: MouseEvent) {
     e.preventDefault();
     const startX = e.clientX;
     const startWidth = columnWidths[column];
     // Define min widths per column
     const minWidths: Record<keyof ColumnWidths, number> = {
       ref: 60,
       graph: Math.max(maxColumns, 1) * 12,  // laneWidth * maxColumns
       author: 60,
       date: 60,
       sha: 50,
     };

     function onMouseMove(ev: MouseEvent) {
       const delta = ev.clientX - startX;
       const newWidth = Math.max(minWidths[column], Math.min(400, startWidth + delta));
       columnWidths = { ...columnWidths, [column]: newWidth };
     }

     function onMouseUp() {
       setColumnWidths(columnWidths);  // persist only on release
       window.removeEventListener('mousemove', onMouseMove);
       window.removeEventListener('mouseup', onMouseUp);
     }

     window.addEventListener('mousemove', onMouseMove);
     window.addEventListener('mouseup', onMouseUp);
   }
   ```
